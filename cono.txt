<script>
var camera, scene, renderer;
var geometria, material, mesh;
var clock;

function init() {
    renderer = new THREE.WebGLRenderer();
    renderer.setSize(window.innerWidth*0.99, window.innerHeight*0.99);
    document.body.appendChild(renderer.domElement);
    scene = new THREE.Scene();
 

     const geometry = new THREE.ConeGeometry( 1.5, 3, 10 );
	const material = new THREE.MeshBasicMaterial( {color: 0xFF5DA2} );
	const cone = new THREE.Mesh( geometry, material);
	scene.add( cone );
	cone.position.set(1,0,0);
        cone.geometry.rotateZ(-Math.PI/2.1);
	
    const axesHelper = new THREE.AxesHelper(10);
    scene.add(axesHelper);

    camera = new THREE.PerspectiveCamera(40, window.innerWidth / window.innerHeight, 0.1, 100);
    camera.position.set(10, 5, 6);
    camera.lookAt(cone.position);
    clock = new THREE.Clock();
    window.addEventListener('resize', onWindowResize, false);
}

var dir = 1;

function onWindowResize() {
    windowHalfX = window.innerWidth / 2;
    windowHalfY = window.innerHeight / 2;
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
}

init();
renderer.render(scene, camera);
